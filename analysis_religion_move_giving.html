Does religiosity buffer the negative effect of moving on giving?

# Research design:
#   - Panel of PSID families (2001-2021, biennial)
#   - Outcome: total charitable giving (sum of donation categories)
#   - Treatment: moved (binary, from "WHY MOVED" reason codes)
#   - Moderator: religious (binary, Religious_Preference != 0/97/98/99)
#   - Model: family fixed effects with moved x religious interaction
#   - Family FE absorbs time-invariant family characteristics
#   - Wave FE absorbs macro trends in giving
#
# NOTE: We keep ALL families including splitoffs and use PSID
# longitudinal family weights (family_weight) to maintain
# representativeness and correct for attrition + splitoff dilution.
# id_1968 tracks the 1968 lineage; family_id is the unit of analysis.

library(tidyverse)
library(arrow)
## Warning: package 'arrow' was built under R version 4.5.2
library(fixest)
## Warning: package 'fixest' was built under R version 4.5.2
library(here)

# ---- 1. Load the family-level panel ----
panel <- read_parquet(here("output", "psid_family_year_2001_latest.parquet"))

cat("Raw panel:", nrow(panel), "rows,", ncol(panel), "columns\n")

cat("Unique id_1968 (lineages):", n_distinct(panel$id_1968), "\n\n")

# ---- 2. Derive key variables ----

# 2a. Total giving: sum available dollar-amount donation columns
donation_cols <- c(
  "DOLLAR AMT OF RELIGIOUS DONATIONS",
  "DOLLAR AMT OF NEEDY DONATIONS",
  "DOLLAR AMT OF HEALTH DONATIONS",
  "DOLLAR AMT OF EDUCATION DONATIONS"
)
available_donation_cols <- donation_cols[donation_cols %in% names(panel)]
cat("Donation columns found:", length(available_donation_cols), "\n")
## Donation columns found: 4
cat("  ", paste(available_donation_cols, collapse = "\n   "), "\n\n")
##    DOLLAR AMT OF RELIGIOUS DONATIONS
##    DOLLAR AMT OF NEEDY DONATIONS
##    DOLLAR AMT OF HEALTH DONATIONS
##    DOLLAR AMT OF EDUCATION DONATIONS
# 2b. Build analysis dataset
#     Clean PSID sentinel codes from dollar amounts:
#     Religious donations use 6+ digit sentinels (999997+)
#     Needy/Health/Education use 5-digit sentinels (99997+)
nonrelig_cols <- c(
  "DOLLAR AMT OF NEEDY DONATIONS",
  "DOLLAR AMT OF HEALTH DONATIONS",
  "DOLLAR AMT OF EDUCATION DONATIONS"
)
relig_cols <- "DOLLAR AMT OF RELIGIOUS DONATIONS"

analysis <- panel |>
  mutate(
    across(
      any_of(nonrelig_cols),
      ~ if_else(.x >= 99997, NA_real_, .x)
    ),
    across(
      any_of(relig_cols),
      ~ if_else(.x >= 999997, NA_real_, .x)
    )
  ) |>
  mutate(
    # --- Outcome: total giving ---
    total_giving = rowSums(
      across(all_of(available_donation_cols),
             ~ replace_na(pmax(.x, 0), 0)),
      na.rm = TRUE
    ),
    log_giving = log1p(total_giving),

    # --- Treatment: whether the family moved since last wave ---
    #     "WHY MOVED 1ST" coded 0 = did not move; 1-9 = reason
    #     97/98/99 are sentinel codes (Other/DK/Refused) — treat
    #     as NA because we can't confirm a move actually occurred
    moved = case_when(
      is.na(`WHY MOVED 1ST`) ~ 0L,
      `WHY MOVED 1ST` >= 97 ~ NA_integer_,
      `WHY MOVED 1ST` > 0 ~ 1L,
      TRUE ~ 0L
    ),

    # --- Moderator: religious affiliation ---
    #     PSID codes: 0 = none, 1-13 = denominations,
    #                 97 = other, 98 = DK, 99 = NA/refused
    religious = case_when(
      is.na(Religious_Preference) ~ NA_integer_,
      Religious_Preference %in% c(0, 97, 98, 99) ~ 0L,
      TRUE ~ 1L
    ),

    # --- Controls ---
    age = as.numeric(head_age),
    age2 = age^2,

    # --- Longitudinal family key ---
    # family_id is recycled across waves (useless for tracking).
    # Construct a composite key: id_1968 + head birth year + sex
    # to uniquely identify the same family across waves.
    birth_year = wave - age,
    family_key = paste(id_1968, birth_year, head_sex, sep = "_")
  ) |>
  select(
    id_1968, family_id, family_key, wave, interview_year,
    family_weight,
    total_giving, log_giving,
    moved, religious,
    Religious_Preference,
    age, age2,
    `Current State`,
    any_of("n_children_u18")
  ) |>
  filter(!is.na(id_1968), !is.na(religious),
         !is.na(family_weight), family_weight > 0)

# Deduplicate: if two families share the same key in a wave
# (e.g., same-sex twins from same lineage), keep oldest head
n_pre <- nrow(analysis)
analysis <- analysis |>
  group_by(family_key, wave) |>
  slice_max(age, n = 1, with_ties = FALSE) |>
  ungroup()
cat("After dedup:", nrow(analysis), "of", n_pre,
    "rows (dropped", n_pre - nrow(analysis), ")\n")

# --- Lagged religiosity (pre-move moderator) ---
# family_key uniquely tracks a family across waves
analysis <- analysis |>
  arrange(family_key, wave) |>
  group_by(family_key) |>
  mutate(religious_lag = lag(religious, order_by = wave)) |>
  ungroup()

# ---- 4. Fixed effects regression (weighted) ----
# family_key FE absorbs time-invariant family characteristics
# (composite key: id_1968 + head birth year + sex)
# wave FE absorbs macro trends in giving
# Weights = PSID longitudinal family weights
# religious_lag = religiosity from the PRIOR wave (pre-move)
#   — avoids endogeneity from moving changing religious ID

# Subset with non-missing lagged religiosity
analysis_lag <- analysis |>
  filter(!is.na(religious_lag), !is.na(moved))

cat("Lagged-religion sample:", nrow(analysis_lag),
    "obs (first wave of each family dropped)\n\n")
## Lagged-religion sample: 69218 obs (first wave of each family dropped)
# Model 1: Does moving reduce giving?
m1 <- feols(log_giving ~ moved | family_key + wave,
            data = analysis_lag,
            weights = ~family_weight)
## NOTE: 3,456/0 fixed-effect singletons were removed (3,456 observations).
# Model 2: Does PRE-MOVE religiosity moderate the effect?
m2 <- feols(
  log_giving ~ moved * religious_lag | family_key + wave,
  data = analysis_lag,
  weights = ~family_weight
)
## NOTE: 3,456/0 fixed-effect singletons were removed (3,456 observations).
# Model 3: Add age controls
m3 <- feols(
  log_giving ~ moved * religious_lag + age + age2 |
    family_key + wave,
  data = analysis_lag,
  weights = ~family_weight
)
## NOTE: 3,456/0 fixed-effect singletons were removed (3,456 observations).
## The variable 'age' has been removed because of collinearity (see $collin.var).
# Cluster standard errors by family
etable(m1, m2, m3,
       headers = c("Baseline", "Interaction", "Controls"),
       vcov = "cluster",
       se.below = TRUE)

# ---- 6c. Event-time plot around the first observed move ----
# For a cleaner descriptive event window, focus on families with exactly
# one observed move and at least one pre- and post-move wave. Each
# family's giving is indexed to the prior wave (t = -2).
first_move_families <- analysis |>
  filter(!is.na(religious_lag), !is.na(moved)) |>
  group_by(family_key) |>
  summarise(
    move_count = sum(moved == 1, na.rm = TRUE),
    first_move_wave = ifelse(
      any(moved == 1, na.rm = TRUE),
      min(wave[moved == 1], na.rm = TRUE),
      NA_real_
    ),
    pre_obs = sum(wave < first_move_wave, na.rm = TRUE),
    post_obs = sum(wave > first_move_wave, na.rm = TRUE),
    pre_move_religious = religious_lag[match(first_move_wave, wave)],
    baseline_log_giving = log_giving[match(first_move_wave - 2, wave)],
    .groups = "drop"
  ) |>
  filter(
    move_count == 1,
    pre_obs >= 1,
    post_obs >= 1,
    !is.na(pre_move_religious),
    !is.na(baseline_log_giving)
  )

event_sample <- analysis |>
  filter(!is.na(religious_lag), !is.na(moved)) |>
  inner_join(
    first_move_families |>
      select(family_key, first_move_wave,
             pre_move_religious, baseline_log_giving),
    by = "family_key"
  ) |>
  mutate(
    event_time = wave - first_move_wave,
    change_from_pre_move = log_giving - baseline_log_giving,
    Religious = factor(
      pre_move_religious,
      levels = c(0, 1),
      labels = c("Not religious", "Religious")
    )
  ) |>
  filter(event_time %in% c(-4, -2, 0, 2, 4))

event_summary <- event_sample |>
  group_by(event_time, Religious) |>
  summarise(
    mean_change = weighted.mean(change_from_pre_move,
                                family_weight, na.rm = TRUE),
    n_families = n_distinct(family_key),
    .groups = "drop"
  )

n_event_families <- n_distinct(event_sample$family_key)

event_time_plot <- ggplot(
  event_summary,
  aes(x = event_time, y = mean_change,
      colour = Religious, group = Religious)
) +
  geom_hline(yintercept = 0, linetype = "dashed", colour = "grey60") +

    breaks = c(-4, -2, 0, 2, 4),
    labels = c("-4", "-2", "0", "+2", "+4")
  ) +
  labs(
    title = "Change in Giving Around the First Observed Move",
    subtitle = paste0(
      "One-time movers only; indexed to the prior wave (t = -2), ",
      "N = ", scales::comma(n_event_families), " families"
    ),
    x = "Years relative to move",
    y = "Change in log(giving + 1) relative to t = -2",
    colour = NULL
  ) +
  theme_minimal(base_size = 13) +
